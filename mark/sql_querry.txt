CREATE TABLE history (
	user_id data_type NOT NULL,
	name data_type NOT NULL,
   	symbole data_type NOT NULL,
	date data_type NOT NULL,
	transaction_id data_type NOT NULL,
	price data_type NUMERIC NOT NULL,
	quantity data_type NUMERIC NOT NULL,
	table_constraints
);


CREATE TABLE portfolio (
	user_id data_type NOT NULL,
   	symbole data_type NOT NULL,
	quantity data_type NUMERIC NOT NULL
);
